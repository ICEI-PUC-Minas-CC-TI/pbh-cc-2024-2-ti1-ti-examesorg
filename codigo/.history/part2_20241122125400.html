function showCalendar() { document.getElementById('calendar').style.display = 'block'; populateTimeSlots(); } function populateTimeSlots() { const timeSlots = [ '08:00', '08:30', '09:00', '09:30', '10:00', '10:30', '11:00', '11:30' ]; const timeSlotsList = document.getElementById('timeSlots'); timeSlotsList.innerHTML = ''; // Clear previous time slots timeSlots.forEach(slot => { const li = document.createElement('li'); li.textContent = slot; timeSlotsList.appendChild(li); }); } function scheduleAppointment() { const selectedDate = document.getElementById('selectedDate').value; const notes = document.getElementById('notes').value; alert(`Appointment scheduled for ${selectedDate} with notes: ${notes}`); }
